```javascript // script.js // Input validation function validateInput() { const input = document.getElementById('text'); const inputValue = input.value.trim(); if (!inputValue) { alert('Please enter a valid text or URL.'); return false; } return true; } // Generate QR code function generateQR() { if (validateInput()) { const input = document.getElementById('text'); const inputValue = input.value.trim(); const qrcode = new QRCode(document.getElementById('qrcode'), { text: inputValue, width: 100, height: 100, colorDark: "#000000", colorLight: "#ffffff", correctLevel: QRCode.CorrectLevel.H }); } } // Add event listener to generate button document.querySelector('button').addEventListener('click', generateQR); ```
